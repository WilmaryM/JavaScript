//Funcion callBack:

function imprimir(mensaje){
    console.log(mensaje);
}
function suma( op1, op2, funcionCallBack){
    let res = op1 + op2;
    funcionCallBack(`Resultado: ${res}`);
}

suma(5,3, imprimir);
